# - set up DHCPv6 server - not required for the lab but here for reference since the lab doesn't have IPv4 access to get the packages required
sudo apt-get install isc-dhcp-server
# - You will then need to edit the configuration file: /etc/dhcp/dhcpd.conf
# - build out a scope
subnet6 <address/prefix>
range6 (<start ip> <end ip> | <network/prefix>)
#
# - To have the DHCP server to start automatically edit the init script in /etc/init.d/isc-dhcp-server
# - Change all the places where dhcpd is started and add the parameter -6

Reference link for Ubuntu:
https://wiki.ubuntu.com/DHCPv6
